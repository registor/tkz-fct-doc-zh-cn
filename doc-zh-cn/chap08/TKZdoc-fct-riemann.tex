\begin{NewMacroBox}{tkzDrawRiemannSum}{\oarg{命令选项}\marg{$f(t)$}}
该命令用于绘制函数的黎曼和矩形，可以使用所有有效\TIKZ{}选项。

\begin{tabular}{lll}
\toprule
选项             & 默认值 & 含义                         \\
\midrule
\TOline{interval}{1:2}{区间长度}
\TOline{number}{10}{子间隔数}
\bottomrule
\end{tabular}

可以组合使用四个相关命令，并为其设置相同选项。
\end{NewMacroBox}

\subsection{求黎曼和}

\begin{tkzexample}[vbox]
\begin{tikzpicture}[scale=3.5]
\tkzInit[xmax=3,ymax=1.75]
\tkzAxeXY
\tkzGrid(0,0)(3,2)
\tkzFct[color = red, domain =1/3:3]{0.125*(3*x-1)+0.375*(3*x-1)/(x*x)}
\tkzDrawRiemannSum[fill=green!40,opacity=.2,color=green,
                   line width=1pt,interval=1./2:exp(1),number=10]
\end{tikzpicture}
\end{tkzexample}

\begin{NewMacroBox}{tkzDrawRiemannSumInf}{\oarg{命令选项}}
是第1个命令的变体，但分割矩形始终位于函数曲线下方。
 \end{NewMacroBox}

 \subsection{\tkzcname{tkzDrawRiemannSumInf}命令示例}

\begin{tkzexample}[vbox]
\begin{tikzpicture}[scale=1.75]
\tkzInit[xmin=-3,xmax=6,ymin=-2,ymax=14,ystep=2]
\tkzDrawX \tkzDrawY
\tkzFct[line width=2pt,color = red, domain =-3:6]{(-\x-2)*(\x-5)}
\tkzDrawRiemannSumInf[fill=green!40,opacity=.5,interval=-1:5,number=10]
\end{tikzpicture}
\end{tkzexample}
